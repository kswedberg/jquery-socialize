jQuery Socialize Plugin

Demo

Simple call to .socialize(). 

$(document).ready(function() {
  $('#social-buttons').socialize();
});

Call to .socialize() with a few options set.

$(document).ready(function() {
  $('#more-social-buttons').socialize({
    sites: ['delicious', 'myspace', 'google'],
    events: {
      click: function(event) {
        $(this).parent().parent().append('<div>' + event.type + ' '+ this.className + ' ' + '</div>');
        window.open(this.href);          
      },
      mouseenter: function() {
        console.log('hi there');
      }
    },
    insertion: 'insertAfter'
  });
});
